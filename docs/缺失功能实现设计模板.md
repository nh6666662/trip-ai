### 1.1 和风天气 API 接入

**目标**：为 Harness Engine 天气维度注入真实数据，天气评分从默认 60 分变为真实值。

**和风天气 API 选型**：

| 接口 | 用途 | 免费额度 |
|---|---|---|
| `/v7/weather/7d` | 7 天预报（行程生成时查询） | 1000 次/日 |
| `/v7/weather/now` | 实时天气（行程中预警检测） | 1000 次/日 |
| `/v7/warning/now` | 灾害预警（暴雨/高温/大风） | 1000 次/日 |

```typescript
/**
 * 和风天气 API 客户端
 * 文档：https://dev.qweather.com/docs/api/weather/

const QWEATHER_KEY = process.env.QWEATHER_API_KEY || '';
const QWEATHER_BASE = 'https://devapi.qweather.com';

export interface QWeatherForecast {
  fxDate: string;       // 预报日期 2026-07-05
  tempMax: string;      // 最高温 °C
  tempMin: string;      // 最低温 °C
  textDay: string;      // 白天天气：晴/多云/小雨/大雨/暴雨/雪
  textNight: string;    // 夜间天气
  windSpeedDay: string; // 白天风速 km/h
  humidity: string;     // 相对湿度 %
  uvIndex: string;      // 紫外线指数
}

export interface QWeatherNow {
  temp: string;         // 当前温度 °C
  text: string;         // 当前天气
  humidity: string;     // 湿度 %
  windSpeed: string;    // 风速 km/h
}

export interface QWeatherWarning {
  title: string;        // 预警标题
  level: string;        // 预警等级：蓝色/黄色/橙色/红色
  type: string;         // 预警类型：暴雨/高温/大风
  text: string;         // 预警详情
}

/**
 * 获取城市 7 天天气预报
 * @param location 城市名或经纬度（"116.41,39.92"）
 */
export async function getWeatherForecast(
  location: string,
): Promise<QWeatherForecast[]> {
  if (!QWEATHER_KEY) {
    console.warn('[Weather] QWEATHER_API_KEY 未配置');
    return [];
  }

  // 先地理编码（城市名 → LocationID）
  const locationId = await resolveLocationId(location);
  if (!locationId) return [];

  const url = `${QWEATHER_BASE}/v7/weather/7d?location=${locationId}&key=${QWEATHER_KEY}&lang=zh`;
  const res = await fetch(url, { next: { revalidate: 3600 } });
  const data = await res.json();

  if (data.code !== '200') {
    console.warn('[Weather] 预报查询失败:', data.code);
    return [];
  }

  return data.daily ?? [];
}

/**
 * 获取实时天气
 */
export async function getWeatherNow(location: string): Promise<QWeatherNow | null> {
  if (!QWEATHER_KEY) return null;
  const locationId = await resolveLocationId(location);
  if (!locationId) return null;

  const url = `${QWEATHER_BASE}/v7/weather/now?location=${locationId}&key=${QWEATHER_KEY}`;
  const res = await fetch(url, { next: { revalidate: 300 } }); // 5 分钟缓存
  const data = await res.json();

  return data.code === '200' ? data.now : null;
}

/**
 * 获取灾害预警
 */
export async function getWeatherWarnings(location: string): Promise<QWeatherWarning[]> {
  if (!QWEATHER_KEY) return [];
  const locationId = await resolveLocationId(location);
  if (!locationId) return [];

  const url = `${QWEATHER_BASE}/v7/warning/now?location=${locationId}&key=${QWEATHER_KEY}&lang=zh`;
  const res = await fetch(url, { next: { revalidate: 300 } });
  const data = await res.json();

  return data.code === '200' ? (data.warning ?? []) : [];
}

/** 城市名 → LocationID（和风天气内部编码） */
async function resolveLocationId(location: string): Promise<string | null> {
  const url = `https://geoapi.qweather.com/v2/city/lookup?location=${encodeURIComponent(location)}&key=${QWEATHER_KEY}&number=1`;
  const res = await fetch(url, { next: { revalidate: 86400 } });
  const data = await res.json();
  return data.code === '200' && data.location?.[0]
    ? data.location[0].id
    : null;
}
```

```typescript
/**
 * 和风天气 → Harness Engine WeatherData 适配器
 * 将和风天气 API 返回格式转换为 scoreWeather() 所需的 WeatherData 格式
 */
import type { WeatherData } from '@/lib/harness/tools';
import type { QWeatherForecast, QWeatherNow } from './index';

/** 将 7 天预报中某一天的数据转换为 WeatherData */
export function forecastToWeatherData(forecast: QWeatherForecast): WeatherData {
  const rainMap: Record<string, number> = {
    '晴': 0, '多云': 0.05, '阴': 0.1,
    '阵雨': 0.4, '小雨': 0.5, '中雨': 0.7,
    '大雨': 0.85, '暴雨': 0.95, '雷阵雨': 0.6,
    '小雪': 0.5, '中雪': 0.7, '大雪': 0.9,
  };

  return {
    condition: forecast.textDay,
    temperature: (parseFloat(forecast.tempMax) + parseFloat(forecast.tempMin)) / 2,
    rainProbability: rainMap[forecast.textDay] ?? 0.3,
    humidity: parseFloat(forecast.humidity) || 50,
    windSpeed: parseFloat(forecast.windSpeedDay) || 10,
  };
}

/** 将实时天气数据转换为 WeatherData */
export function nowToWeatherData(now: QWeatherNow): WeatherData {
  const rainMap: Record<string, number> = {
    '晴': 0, '多云': 0.05, '阴': 0.1,
    '小雨': 0.5, '中雨': 0.7, '大雨': 0.85, '暴雨': 0.95,
  };

  return {
    condition: now.text,
    temperature: parseFloat(now.temp) || 20,
    rainProbability: rainMap[now.text] ?? 0.3,
    humidity: parseFloat(now.humidity) || 50,
    windSpeed: parseFloat(now.windSpeed) || 10,
  };
}
```

```typescript
/**
 * 行程导出工具
 * MVP 方案：html2canvas 截图 + jspdf 生成 PDF
 */
import type { Trip, TripNode } from '@/types/database';

export interface ExportOptions {
  trip: Trip;
  nodes: TripNode[];
  format: 'pdf' | 'image' | 'print';
}

/**
 * 导出行程为 PDF
 * 原理：渲染隐藏的 HTML 模板 → html2canvas 截图 → jspdf 组装 PDF
 */
export async function exportTripToPDF(options: ExportOptions): Promise<void> {
  const { trip, nodes } = options;

  // 动态导入（避免 SSR 时加载 canvas 库）
  const html2canvas = (await import('html2canvas')).default;
  const { jsPDF } = await import('jspdf');

  // 创建导出容器
  const container = document.createElement('div');
  container.style.cssText = 'position:fixed;left:-9999px;top:0;width:794px;background:white;padding:40px;font-family:sans-serif;';
  container.innerHTML = buildPDFHTML(trip, nodes);
  document.body.appendChild(container);

  try {
    const canvas = await html2canvas(container, { scale: 2, useCORS: true });
    const pdf = new jsPDF('p', 'mm', 'a4');
    const imgWidth = 210; // A4 宽度 mm
    const imgHeight = (canvas.height * imgWidth) / canvas.width;
    const imgData = canvas.toDataURL('image/png');

    // 多页处理
    let heightLeft = imgHeight;
    let position = 0;
    pdf.addImage(imgData, 'PNG', 0, position, imgWidth, imgHeight);
    heightLeft -= 297; // A4 高度 mm

    while (heightLeft > 0) {
      position = -297 * Math.ceil((imgHeight - heightLeft) / 297);
      pdf.addPage();
      pdf.addImage(imgData, 'PNG', 0, position, imgWidth, imgHeight);
      heightLeft -= 297;
    }

    pdf.save(`${trip.destination}-${trip.start_date}-行程.pdf`);
  } finally {
    document.body.removeChild(container);
  }
}

/** 构建 PDF HTML 模板 */
function buildPDFHTML(trip: Trip, nodes: TripNode[]): string {
  const nodesByDay = groupByDay(nodes);
  return `
    <div style="text-align:center;margin-bottom:30px;">
      <h1 style="color:#6366F1;font-size:28px;margin:0;">${trip.destination} 行程</h1>
      <p style="color:#737373;font-size:14px;">
        ${trip.start_date} ~ ${trip.end_date} · ${trip.traveler_count}人 · ${trip.pace === 'tight' ? '紧凑版' : '松弛版'}
      </p>
    </div>
    ${Object.entries(nodesByDay).map(([day, dayNodes]) => `
      <div style="margin-bottom:24px;">
        <h2 style="color:#262626;font-size:18px;border-bottom:2px solid #E5E5E5;padding-bottom:8px;">
          第 ${day} 天
        </h2>
        ${dayNodes.map((node) => `
          <div style="display:flex;gap:12px;padding:8px 0;border-bottom:1px solid #F5F5F5;">
            <span style="color:#6366F1;font-weight:600;min-width:50px;">
              ${formatTime(node.start_time)}
            </span>
            <div>
              <div style="font-weight:500;color:#262626;">${node.name}</div>
              <div style="color:#737373;font-size:12px;">
                ${node.duration_minutes}分钟${node.transit_minutes ? ` · 交通${node.transit_minutes}分钟` : ''}
              </div>
            </div>
          </div>
        `).join('')}
      </div>
    `).join('')}
    <div style="text-align:center;color:#A3A3A3;font-size:11px;margin-top:40px;">
      由旅智 TripAI 生成 · ${new Date().toLocaleDateString('zh-CN')}
    </div>
  `;
}

function groupByDay(nodes: TripNode[]): Record<number, TripNode[]> {
  const groups: Record<number, TripNode[]> = {};
  nodes.forEach((node) => {
    const day = (node.metadata as any)?.day ?? 1;
    if (!groups[day]) groups[day] = [];
    groups[day].push(node);
  });
  return groups;
}

function formatTime(iso: string): string {
  const d = new Date(iso);
  return `${d.getHours().toString().padStart(2, '0')}:${d.getMinutes().toString().padStart(2, '0')}`;
}

/** 浏览器打印（最小依赖方案） */
export function printTrip(trip: Trip, nodes: TripNode[]): void {
  const printWindow = window.open('', '_blank');
  if (!printWindow) return;
  printWindow.document.write(`
    <!DOCTYPE html><html><head><title>${trip.destination} 行程</title>
    <style>
      body { font-family: sans-serif; padding: 40px; color: #262626; }
      h1 { color: #6366F1; text-align: center; }
      .subtitle { color: #737373; text-align: center; }
      .day-title { border-bottom: 2px solid #E5E5E5; padding-bottom: 8px; }
      .node { display: flex; gap: 12px; padding: 6px 0; border-bottom: 1px solid #F5F5F5; }
      .time { color: #6366F1; font-weight: 600; min-width: 50px; }
      @media print { body { padding: 20px; } }
    </style></head><body>${buildPDFHTML(trip, nodes)}</body></html>
  `);
  printWindow.document.close();
  printWindow.print();
}
```

```typescript
'use client';

import { useState, useEffect, useCallback } from 'react';

export interface GeolocationState {
  latitude: number | null;
  longitude: number | null;
  accuracy: number | null;
  error: string | null;
  loading: boolean;
}

/**
 * 浏览器 GPS 定位 Hook
 * 封装 navigator.geolocation，处理权限/错误/超时
 */
export function useGeolocation(options?: PositionOptions) {
  const [state, setState] = useState<GeolocationState>({
    latitude: null,
    longitude: null,
    accuracy: null,
    error: null,
    loading: true,
  });

  const requestLocation = useCallback(() => {
    if (!navigator.geolocation) {
      setState(s => ({ ...s, error: '浏览器不支持定位', loading: false }));
      return;
    }

    setState(s => ({ ...s, loading: true, error: null }));

    navigator.geolocation.getCurrentPosition(
      (pos) => {
        setState({
          latitude: pos.coords.latitude,
          longitude: pos.coords.longitude,
          accuracy: pos.coords.accuracy,
          error: null,
          loading: false,
        });
      },
      (err) => {
        setState(s => ({
          ...s,
          error: err.code === 1 ? '用户拒绝定位权限' : err.message,
          loading: false,
        }));
      },
      {
        enableHighAccuracy: false,
        timeout: 10000,
        maximumAge: 300000, // 5 分钟缓存
        ...options,
      },
    );
  }, [options]);

  // 组件挂载时自动请求
  useEffect(() => { requestLocation(); }, [requestLocation]);

  return { ...state, refresh: requestLocation };
}
```

```typescript
export function useNearbyRecommendations() {
  const { latitude, longitude, error: geoError } = useGeolocation();

  const { data: nearbySpots = [], isLoading } = useQuery({
    queryKey: [...queryKeys.spots('nearby'), latitude, longitude],
    queryFn: () =>

        `/api/ugc/spots?lat=${latitude}&lng=${longitude}&radius=5000&limit=8`,

    staleTime: 5 * 60 * 1000, // 5 分钟缓存
  });

  return { nearbySpots, isLoading, geoError, hasLocation: !!latitude };
}
```

新增 PostgreSQL RPC 函数：

```sql
CREATE OR REPLACE FUNCTION nearby_spots(lat DOUBLE PRECISION, lng DOUBLE PRECISION, radius_meters INT)
RETURNS TABLE(id UUID, name TEXT, rating FLOAT, distance_km FLOAT) AS $$
BEGIN
  RETURN QUERY
  SELECT
    s.id,
    s.name,
    s.rating,
    ROUND(
      (6371 * acos(
        cos(radians(lat)) * cos(radians(s.latitude)) *
        cos(radians(s.longitude) - radians(lng)) +
        sin(radians(lat)) * sin(radians(s.latitude))
      ))::numeric, 2
    )::FLOAT AS distance_km
  FROM spots s
  WHERE 6371 * acos(
    cos(radians(lat)) * cos(radians(s.latitude)) *
    cos(radians(s.longitude) - radians(lng)) +
    sin(radians(lat)) * sin(radians(s.latitude))
  ) < radius_meters / 1000.0
  ORDER BY distance_km
  LIMIT 20;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;
```

### 2.2 预警自动触发机制

**目标**：定时检测天气/交通/UGC 异常，自动生成预警写入 `realtime_alerts` 表，通过 Supabase Realtime 推送到前端。

#### 🆕 新建文件：`supabase/functions/alert-generator/index.ts`

```typescript
/**
 * 预警自动生成 Edge Function
 * 由 Supabase Cron 每 30 分钟触发一次
 *
 * 检测逻辑：
 * 1. 查询所有 status='confirmed' 的活跃行程
 * 2. 对每个行程检查天气预警 + 景点人流异常
 * 3. 高置信度异常 → INSERT realtime_alerts（触发 Realtime 推送）
 */

import { createClient } from 'https://esm.sh/@supabase/supabase-js@2';

Deno.serve(async (req) => {
  // 验证 Cron 调用密钥
  const authHeader = req.headers.get('Authorization');
  if (authHeader !== `Bearer ${Deno.env.get('CRON_SECRET')}`) {
    return new Response('Unauthorized', { status: 401 });
  }

  const supabase = createClient(
    Deno.env.get('SUPABASE_URL')!,
    Deno.env.get('SUPABASE_SERVICE_ROLE_KEY')!,
  );

  // 查询当前活跃行程（今天及未来 7 天）
  const today = new Date().toISOString().split('T')[0];
  const { data: activeTrips } = await supabase
    .from('trips')
    .select('id, destination, start_date')
    .eq('status', 'confirmed')
    .gte('end_date', today)
    .lte('start_date', today);

  if (!activeTrips?.length) {
    return new Response(JSON.stringify({ checked: 0, alerts: 0 }));
  }

  let alertsCreated = 0;

  for (const trip of activeTrips) {
    // ===== 检测 1：天气预警 =====
    const warnings = await checkWeatherWarnings(trip.destination);
    for (const warning of warnings) {
      const existing = await supabase
        .from('realtime_alerts')
        .select('id')
        .eq('trip_id', trip.id)
        .eq('alert_type', 'weather')
        .eq('status', 'pending')
        .maybeSingle();

      if (!existing.data) {
        await supabase.from('realtime_alerts').insert({
          trip_id: trip.id,
          alert_type: 'weather',
          priority: warning.level === '红色' ? 'high' : 'medium',
          title: warning.title,
          description: warning.text,
          confidence: 0.9,
          suggestion: '建议将户外行程调整为室内活动',
        });
        alertsCreated++;
      }
    }

    // ===== 检测 2：景点人流异常 =====
    const crowdAlerts = await checkCrowdAnomalies(trip.id, supabase);
    alertsCreated += crowdAlerts;
  }

  return new Response(JSON.stringify({ checked: activeTrips.length, alerts: alertsCreated }));
});

async function checkWeatherWarnings(destination: string) {
  // 调用和风天气灾害预警 API
  const key = Deno.env.get('QWEATHER_API_KEY');
  if (!key) return [];
  // ... 实现同 src/lib/weather/index.ts 的 getWeatherWarnings
  return [];
}

async function checkCrowdAnomalies(tripId: string, supabase: any): Promise<number> {
  // 查询该行程景点最近 1 小时的 UGC 上报
  const oneHourAgo = new Date(Date.now() - 3600000).toISOString();
  const { data: reports } = await supabase
    .from('ugc_reports')
    .select('spot_id, content')
    .eq('status', 'verified')
    .gte('created_at', oneHourAgo)
    .in('spot_id',
      (await supabase.from('trip_nodes').select('spot_id').eq('trip_id', tripId)).data?.map((n: any) => n.spot_id) ?? []
    );

  if (!reports?.length) return 0;

  // 统计每个景点的上报次数
  const spotCounts = new Map<string, number>();
  reports.forEach((r: any) => {
    spotCounts.set(r.spot_id, (spotCounts.get(r.spot_id) ?? 0) + 1);
  });

  let created = 0;
  for (const [spotId, count] of spotCounts) {
    if (count >= 3) { // 3 人以上报告拥挤
      const existing = await supabase
        .from('realtime_alerts')
        .select('id')
        .eq('trip_id', tripId)
        .eq('alert_type', 'crowd')
        .eq('status', 'pending')
        .maybeSingle();

      if (!existing.data) {
        await supabase.from('realtime_alerts').insert({
          trip_id: tripId,
          alert_type: 'crowd',
          priority: 'medium',
          title: `景点当前人流较密集（${count} 人近 1 小时内上报）`,
          confidence: Math.min(0.5 + count * 0.1, 0.9),
          suggestion: '可考虑调整至其他时段或替换为备选景点',
        });
        created++;
      }
    }
  }
  return created;
}
```

```typescript
'use client';

import { useState, useEffect, useCallback } from 'react';

type Theme = 'light' | 'dark' | 'system';

export function useTheme() {
  const [theme, setThemeState] = useState<Theme>('system');
  const [resolvedTheme, setResolvedTheme] = useState<'light' | 'dark'>('light');

  useEffect(() => {
    // 从 cookie 读取保存的主题
    const saved = document.cookie.match(/theme=(light|dark|system)/)?.[1] as Theme || 'system';
    setThemeState(saved);
  }, []);

  useEffect(() => {
    const root = document.documentElement;
    const isDark = theme === 'dark' ||
      (theme === 'system' && window.matchMedia('(prefers-color-scheme: dark)').matches);

    root.classList.toggle('dark', isDark);
    setResolvedTheme(isDark ? 'dark' : 'light');
  }, [theme]);

  const setTheme = useCallback((t: Theme) => {
    setThemeState(t);
    document.cookie = `theme=${t};path=/;max-age=${365 * 86400}`;
  }, []);

  return { theme, resolvedTheme, setTheme };
}
```

```tsx
'use client';

import * as React from 'react';
import AMapLoader from '@amap/amap-jsapi-loader';
import type { TripNode } from '@/types/database';

interface TripMapProps {
  nodes: TripNode[];
  selectedNodeId?: string | null;
  onNodeClick?: (nodeId: string) => void;
  className?: string;
}

/**
 * 高德地图行程可视化组件
 * 在地图上标记景点位置，连线显示游览顺序
 */
export function TripMap({ nodes, selectedNodeId, onNodeClick, className }: TripMapProps) {
  const mapRef = React.useRef<HTMLDivElement>(null);
  const mapInstance = React.useRef<any>(null);

  React.useEffect(() => {
    if (!mapRef.current || nodes.length === 0) return;

    AMapLoader.load({
      key: process.env.NEXT_PUBLIC_AMAP_KEY!,
      version: '2.0',
      plugins: ['AMap.Marker', 'AMap.Polyline', 'AMap.InfoWindow'],
    }).then((AMap) => {
      const map = new AMap.Map(mapRef.current, {
        zoom: 12,
        viewMode: '2D',
      });
      mapInstance.current = map;

      // 添加景点标记
      const markers = nodes
        .filter(n => n.node_type === 'spot')
        .map((node, idx) => {
          const spot = node as any;
          if (!spot.latitude || !spot.longitude) return null;

          const marker = new AMap.Marker({
            position: [spot.longitude, spot.latitude],
            title: node.name,
            label: {
              content: `<span style="background:#6366F1;color:white;padding:2px 8px;border-radius:12px;font-size:12px;">${idx + 1}</span>`,
              direction: 'top',
            },
            extData: node.id,
          });

          marker.on('click', () => onNodeClick?.(node.id));
          map.add(marker);
          return marker;
        })
        .filter(Boolean);

      // 连线
      const spotNodes = nodes.filter(n => n.node_type === 'spot');
      if (spotNodes.length >= 2) {
        const path = spotNodes
          .filter(n => {
            const s = n as any;
            return s.latitude && s.longitude;
          })
          .map(n => {
            const s = n as any;
            return [s.longitude, s.latitude];
          });

        const polyline = new AMap.Polyline({
          path,
          strokeColor: '#6366F1',
          strokeWeight: 3,
          strokeOpacity: 0.7,
          lineJoin: 'round',
          showDir: true,
        });
        map.add(polyline);
      }

      // 自适应视野
      if (markers.length > 0) {
        map.setFitView(markers, false, [60, 60, 60, 60]);
      }
    });

    return () => {
      mapInstance.current?.destroy();
    };
  }, [nodes]);

  // 选中节点时居中
  React.useEffect(() => {
    if (!selectedNodeId || !mapInstance.current) return;
    const node = nodes.find(n => n.id === selectedNodeId) as any;
    if (node?.latitude && node?.longitude) {
      mapInstance.current.panTo([node.longitude, node.latitude]);
    }
  }, [selectedNodeId, nodes]);

  return (
    <div
      ref={mapRef}
      className={className || 'h-[400px] w-full rounded-lg border border-neutral-200'}
    />
  );
}
```
